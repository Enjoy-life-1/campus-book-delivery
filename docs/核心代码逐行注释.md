def _load_env_file():
    """启动前加载 .env 到 os.environ"""
    root = os.path.dirname(os.path.abspath(__file__))  # 项目根目录（app.py 所在目录）
    try:
        from dotenv import load_dotenv                 # 优先用 python-dotenv 库
        load_dotenv(os.path.join(root, '.env'), override=False)  # 不覆盖已有系统环境变量
        return
    except ImportError:
        pass                                          # 未安装 dotenv 则走下方手动解析
    env_path = os.path.join(root, '.env')
    if not os.path.isfile(env_path):                   # 无 .env 文件则跳过
        return
    try:
        with open(env_path, encoding='utf-8') as f:
            for line in f:
                line = line.strip()                    # 去掉首尾空白
                if not line or line.startswith('#') or '=' not in line:
                    continue                           # 跳过空行、注释、无等号行
                key, val = line.split('=', 1)          # 只按第一个 = 分割
                key, val = key.strip(), val.strip().strip('"').strip("'")  # 去引号
                if key and key not in os.environ:
                    os.environ[key] = val              # 仅当系统未设置时才写入
    except OSError:
        pass                                          # 读文件失败则静默忽略


_load_env_file()                                       # 模块加载时立即执行，早于 Flask 创建

app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False  # 关闭修改追踪，省内存

_cors_origins = os.environ.get('CORS_ORIGINS', '').strip()  # 生产跨域白名单

register_engineering(app)                              # 注册 /api/docs、/api/openapi.json
cache.init_app(app)                                    # 初始化 Flask-Caching
from services.engineering import build_cache_config as _build_cache_config
print(f'[INFO] 缓存后端: {_build_cache_config().get("CACHE_TYPE", "SimpleCache")}')


@app.after_request
def add_security_headers(response):
    response.headers['X-Content-Type-Options'] = 'nosniff'  # 防 MIME 嗅探攻击
    return response


@app.before_request
def _sync_admin_session_flag():
    """每次请求同步 session 中的 is_admin 与数据库一致"""
    if 'user_id' not in session:
        return                                         # 未登录不处理
    flag = is_admin()                                  # 查库/缓存判断当前是否管理员
    if session.get('is_admin') != flag:
        session['is_admin'] = flag                     # 不一致则更新 session
        session.modified = True                        # 标记 session 已修改需写回 Cookie


@app.before_request
def _guard_admin_api():
    """拦截非管理员的 /api/admin/* 请求（登录接口除外）"""
    path = request.path or ''
    if not path.startswith('/api/admin/') or path == '/api/admin/login':
        return                                         # 非管理 API 或登录接口放行
    if request.method == 'OPTIONS':
        return                                         # CORS 预检放行
    if not is_admin():
        return jsonify({'status': 'error', 'message': '无管理权限'}), 403


@app.route('/static/<path:filename>')

def bootstrap_database():
    """启动时建库、迁移、种子用户"""
    if get_sqlalchemy_uri().startswith('mysql'):
        from database_config import ensure_mysql_database
        db_name = ensure_mysql_database()              # MySQL 库不存在则 CREATE DATABASE
        print(f'[INFO] MySQL database ready: {db_name}')
    with app.app_context():                            # 进入 Flask 应用上下文（ORM 需要）
        db.create_all()                                # 按 models 建表（缺表时）
        try:
            from upgrade_db import upgrade
            upgrade()                                  # 执行 upgrade_db 脚本迁移
        except Exception as e:
            print(f'[WARN] upgrade_db skipped: {e}')
        try:
            from db_migrate import run_migrations
            run_migrations()                           # 执行 Alembic 迁移
        except Exception as e:
            print(f'[WARN] alembic migrate skipped: {e}')
        if User.query.filter_by(username='admin').first():
            return                                     # 已有 admin 则跳过种子数据
        db.session.add(User(                           # 默认管理员
            id='1', username='admin', password=hash_password('admin123'),
            email='admin@example.com', phone='13800138000',
            created_at=time.strftime('%Y-%m-%d %H:%M:%S'), is_admin=True,
        ))
        db.session.add(User(                           # 默认测试学生账号
            id='2', username='student1', password=hash_password('student123'),
            email='student1@example.com', phone='13900139000',
            created_at=time.strftime('%Y-%m-%d %H:%M:%S'), is_admin=False,
        ))
        db.session.commit()
        print('[INFO] created default users: admin / student1')
    check_security_warnings()                          # 生产环境安全检查


def check_security_warnings():
    if os.environ.get('FLASK_ENV', '').lower() != 'production':
        return                                         # 非生产不检查
    with app.app_context():
        admin = User.query.filter_by(username='admin').first()
        if admin and verify_password('admin123', admin.password):
            print('[WARN] 生产环境 admin 仍为默认密码 admin123，请立即修改')
        if not sms_channel_configured():
            print('[WARN] 生产环境未配置短信（阿里云 SMS 或 SMS_WEBHOOK_URL），注册/找回密码不可用')


def check_database():
    """探测数据库是否可连接"""
    from sqlalchemy import text
    uri = get_sqlalchemy_uri()
    with app.app_context():
        try:
            db.session.execute(text('SELECT 1'))       # 简单探活 SQL
            shown = uri.split('@')[-1] if '@' in uri else uri  # 日志隐藏密码部分
            print(f'[INFO] database OK: {shown}')
            return True
        except Exception as e:
            print(f'[ERROR] database connection failed: {e}')
            print('[HINT] 请运行 本机部署.bat 或检查 .env 数据库配置')
            print('[HINT] start MySQL and fix MYSQL_PASSWORD in .env')
            return False


if __name__ == '__main__':                             # 直接 python app.py 时执行
    bootstrap_database()                               # 初始化数据库
    if not check_database():
        raise SystemExit(1)                            # 连不上库则退出
    app.run(debug=True, host='0.0.0.0', port=5000)   # 开发服务器，监听所有网卡 5000 端口

## 7. security.py（密码与登录限流）

```python
"""密码哈希与校验（兼容旧 MD5，登录后自动升级）"""
import hashlib                                   # 兼容旧 MD5 密码
import os
import time
from collections import defaultdict               # IP+用户 → 失败时间列表
from threading import Lock                       # 内存限流线程锁
from werkzeug.security import check_password_hash, generate_password_hash

_login_failures = defaultdict(list)              # 无 Redis 时的失败记录
_login_lock = Lock()
_LOGIN_MAX_ATTEMPTS = 10                         # 5 分钟内最多失败次数
_LOGIN_WINDOW_SEC = 300
_redis = None                                    # Redis 客户端单例
_redis_checked = False


def _get_redis():
    global _redis, _redis_checked
    if _redis_checked:
        return _redis                              # 已探测过则直接返回
    _redis_checked = True
    url = os.environ.get('REDIS_URL', '').strip()
    if not url:
        return None
    try:
        import redis
        _redis = redis.from_url(url, decode_responses=True)
        _redis.ping()                              # 探活
    except Exception:
        _redis = None                              # 连不上则降级内存
    return _redis


def _redis_login_key(username=''):
    return f'login_fail:{_login_rate_key(username)}'


def hash_password(password):
    return generate_password_hash((password or '').strip(), method='pbkdf2:sha256')


def verify_password(raw_password, stored_hash):
    raw = (raw_password or '').strip()
    stored = stored_hash or ''
    if stored.startswith('pbkdf2:') or stored.startswith('scrypt:'):
        return check_password_hash(stored, raw)    # 新格式
    return hashlib.md5(raw.encode()).hexdigest() == stored  # 旧 MD5


def password_needs_upgrade(stored_hash):
    s = stored_hash or ''
    return not (s.startswith('pbkdf2:') or s.startswith('scrypt:'))


def _login_rate_key(username=''):
    try:
        from flask import request
        ip = request.remote_addr or 'unknown'    # 客户端 IP
    except RuntimeError:
        ip = 'unknown'                             # 无请求上下文
    user = (username or '').strip().lower()
    return f'{ip}:{user}' if user else ip        # 组合键防爆破


def check_login_rate_limit(username=''):
    """登录前检查是否触发限流"""
    r = _get_redis()
    if r:
        try:
            return int(r.get(_redis_login_key(username)) or 0) < _LOGIN_MAX_ATTEMPTS
        except Exception:
            pass
    key = _login_rate_key(username)
    now = time.time()
    with _login_lock:
        times = [t for t in _login_failures.get(key, []) if now - t < _LOGIN_WINDOW_SEC]
        _login_failures[key] = times               # 清理过期记录
        if len(times) >= _LOGIN_MAX_ATTEMPTS:
            return False
    return True


def record_login_failure(username=''):
    r = _get_redis()
    if r:
        try:
            key = _redis_login_key(username)
            n = r.incr(key)                        # 原子递增
            if n == 1:
                r.expire(key, _LOGIN_WINDOW_SEC) # 首次设置 TTL
            return
        except Exception:
            pass
    key = _login_rate_key(username)
    with _login_lock:
        _login_failures[key].append(time.time())


def clear_login_attempts(username=''):
    r = _get_redis()
    if r:
        try:
            r.delete(_redis_login_key(username))
        except Exception:
            pass
    key = _login_rate_key(username)
    with _login_lock:
        _login_failures.pop(key, None)             # 登录成功清零
```

import { authAPI } from '@/utils/api'            // 封装好的 /api 请求
import { useUserStore } from '@/stores/user'      // Pinia 用户状态

/** 从服务端 Session 同步用户信息（唯一权威来源） */
export async function refreshSession() {
  const store = useUserStore()                   // 获取 store 实例
  try {
    const res = await authAPI.getUserInfo()      // GET /api/user/info + Cookie
    if (res.status === 'success' && res.user) {
      store.setUser(res.user)                    // 写入 Pinia + localStorage
      return res.user
    }
  } catch (_) {                                   // 401 或未登录
    store.setUser(null)                          // 清空本地状态
  }
  return null
}

export async function loginAndSync(credentials) {
  const res = await authAPI.login(credentials)   // POST /api/login
  if (res.status === 'success') {
    const user = await refreshSession()          // 登录后再拉完整用户信息
    return { ...res, user: user || res.user }
  }
  return res                                     // 失败原样返回给页面展示
}

export async function adminLoginAndSync(credentials) {
  const res = await authAPI.adminLogin(credentials)  // POST /api/admin/login
  if (res.status === 'success') {
    const user = await refreshSession()
    return { ...res, user: user || res.user }
  }

def _post_sms(recipient, title, content, template_params=None):
    from services.sms_aliyun import aliyun_sms_configured, send_verify_code, send_notify_sms
    import re

    hook = _gateway_url('sms')
    prefer_webhook = os.environ.get('SMS_PREFER_WEBHOOK', '1').lower() not in ('0', 'false', 'no')

    def _try_webhook():
        if not _valid_webhook_url(hook):
            return None
        payload = json.dumps({'phone': recipient, 'title': title, 'content': content}).encode('utf-8')
        try:
            req = urllib.request.Request(
                hook, data=payload,
                headers={'Content-Type': 'application/json'}, method='POST',
            )
            urllib.request.urlopen(req, timeout=8)
            return 'sent', ''
        except Exception as e:
            return 'failed', str(e)

    if prefer_webhook and _valid_webhook_url(hook):
        r = _try_webhook()
        if r:
            return r                               # 开发优先 Webhook

    if aliyun_sms_configured():
        params = dict(template_params or {})
        if not params.get('code') and title == '验证码':
            m = re.search(r'(\d{6})', content or '')
            if m:
                params['code'] = m.group(1)
        if params.get('code'):
            return send_verify_code(recipient, params['code'])
        st, detail = send_notify_sms(recipient, content)
        if st == 'sent':
            return st, detail
        if title == '验证码':
            return st, detail

    if not prefer_webhook and _valid_webhook_url(hook):
        r = _try_webhook()
        if r:
            return r

    return 'simulated', '未配置短信通道'

        apply_price_drop_ladders(notify_price_drop)    # 处理到期的阶梯降价
        page = int(request.args.get('page', 1))
        page_size = int(request.args.get('page_size', 1000))
        category = request.args.get('category')
        search = request.args.get('search', '').strip().lower()
        include_sold = request.args.get('include_sold', 'false').lower() == 'true'
        owner_id = request.args.get('owner_id')

        query = Book.query                             # SQLAlchemy 查询起点

        if owner_id and is_logged_in() and str(owner_id) == str(session.get('user_id')):
            query = query.filter_by(owner_id=str(owner_id))  # 我的书：含已售
        elif not include_sold and not is_admin():
            query = query.filter_by(status='available')      # 公开列表只看在售

        if category and category != 'all':
            query = query.filter_by(category=category)

        if search:
            sf = build_search_filter(search)           # 标题/ISBN/作者 OR 条件
            if sf is not None:
                query = query.filter(sf)

        all_books = [book.to_dict() for book in query.all()]

        def get_sort_time(book):
            time_str = book.get('created_at') or book.get('createTime') or book.get('publish_date') or ''
            if not time_str:
                return '1970-01-01 00:00:00'
            if len(time_str) == 10:
                time_str += ' 00:00:00'
            return time_str

        sort_by = request.args.get('sort_by', 'newest')
        if sort_by == 'oldest':
            all_books.sort(key=get_sort_time)
        elif sort_by == 'price_asc':
            all_books.sort(key=lambda b: float(b.get('price') or 0))
        elif sort_by == 'price_desc':
            all_books.sort(key=lambda b: float(b.get('price') or 0), reverse=True)
        elif sort_by == 'title':
            all_books.sort(key=lambda b: (b.get('title') or '').lower())
        else:
            all_books.sort(key=get_sort_time, reverse=True)

        time_range = request.args.get('time_range') or request.args.get('timeRange')
        if time_range and time_range != 'all':
            now = datetime.now()
            filtered_by_time = []
            for book in all_books:
                time_str = book.get('created_at') or book.get('createTime') or book.get('publish_date') or ''
                if not time_str:
                    continue
                try:
                    if len(time_str) == 10:
                        book_dt = datetime.strptime(time_str, '%Y-%m-%d')
                    else:
                        book_dt = datetime.strptime(time_str[:19], '%Y-%m-%d %H:%M:%S')
                except ValueError:
                    continue
                if time_range == 'today' and book_dt.date() == now.date():
                    filtered_by_time.append(book)
                elif time_range == 'week' and book_dt >= now - timedelta(days=7):
                    filtered_by_time.append(book)
                elif time_range == 'month' and book_dt >= now - timedelta(days=30):
                    filtered_by_time.append(book)
            all_books = filtered_by_time

        price_range = request.args.get('priceRange')
        if price_range and price_range != 'all':
            filtered_books = []
            for book in all_books:
                price = float(book.get('price', 0))
                if price_range == '0-50' and 0 <= price <= 50:
                    filtered_books.append(book)
                elif price_range == '50-100' and 50 < price <= 100:
                    filtered_books.append(book)
                elif price_range == '100+' and price > 100:
                    filtered_books.append(book)
            all_books = filtered_books

        all_books = filter_books_campus(all_books, request.args)  # 同校/楼栋/课程等

        start = (page - 1) * page_size
        end = start + page_size
        paginated_books = all_books[start:end]

        return jsonify({
            'status': 'success',
            'books': paginated_books,
            'total': len(all_books),
            'page': page,
            'page_size': page_size
        })

## 24. services/messaging.py（WebSocket 私信推送）

```python
"""私信 WebSocket 推送（flask-sock）"""
import json
import threading

from flask import session
from flask_sock import Sock

sock = Sock()                                    # flask-sock 全局实例
_lock = threading.Lock()                         # 保护 _peers 字典
_peers = {}                                      # ws连接 → {user_id, conv_ids}


def register_chat_sock(app):
    sock.init_app(app)                           # 绑定到 Flask app

    @sock.route('/ws/chat')                      # 前端 ws://host/ws/chat
    def chat_socket(ws):
        if 'user_id' not in session:             # 握手时校验 Session Cookie
            ws.send(json.dumps({'type': 'error', 'message': '未登录'}))
            return
        uid = str(session['user_id'])
        subs = set()                             # 本连接订阅的 conv_id 集合
        with _lock:
            _peers[ws] = {'user_id': uid, 'conv_ids': subs}
        try:
            ws.send(json.dumps({'type': 'connected', 'user_id': uid}))
            while True:
                raw = ws.receive()               # 阻塞收客户端消息
                if raw is None:
                    break                        # 连接关闭
                msg = json.loads(raw)
                mtype = msg.get('type')
                if mtype == 'subscribe':         # 进入某会话页时订阅
                    subs.add(str(msg.get('conv_id', '')))
                elif mtype == 'unsubscribe':     # 离开会话页取消订阅
                    subs.discard(str(msg.get('conv_id', '')))
                elif mtype == 'ping':            # 心跳
                    ws.send(json.dumps({'type': 'pong'}))
        finally:
            with _lock:
                _peers.pop(ws, None)             # 断开清理


def push_chat_event(conv_id, payload, exclude_user=None):
    """HTTP 发消息后调用，推给在线订阅者"""
    data = json.dumps(payload, ensure_ascii=False)
    cid = str(conv_id)
    with _lock:
        targets = [
            (ws, info) for ws, info in list(_peers.items())
            if cid in info.get('conv_ids', set())
            and info.get('user_id') != exclude_user  # 不发回发送者
        ]
    for ws, _ in targets:
        try:
            ws.send(data)                        # type: message / read / ...
        except Exception:
            pass                                 # 连接已断则忽略
```

**流程**：发消息 HTTP POST → 存库 → `push_chat_event` → 对方 WebSocket 实时收到。
